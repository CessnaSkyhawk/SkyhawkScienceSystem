//Telemetry Report// - originally from Probes Before Crew by way of Kiwi Tech Tree
EXPERIMENT_DEFINITION
{
	id = sss_telemetryReport
	title = Telemetry Report
	baseValue = 12
	scienceCap = 12
	dataScale = 1
	requireAtmosphere = False
	situationMask = 63
	biomeMask = 1
	
	RESULTS
	{
		default = The probe's onboard computers have successfully recorded current telemetry data.
		default = Sensors are functioning nominally. Telemetry data logged!
		default = Current location verified, data stored and awaiting transmission.
	}
}
@PART[*]:HAS[@MODULE[ModuleCommand],#vesselType[Probe]]:AFTER[zzzSkyhawkScienceSystem]
{
	MODULE
	{
		name = ModuleScienceExperiment	
		experimentID = sss_telemetryReport
		experimentActionName = Telemetry Report
		resetActionName = Discard Telemetry
		reviewActionName = Review Telemetry
		useStaging = False	
		useActionGroups = True
		hideUIwhenUnavailable = True	
		rerunnable = True
		xmitDataScalar = 1.0
		%sss_preserveScience = True
	}
}
